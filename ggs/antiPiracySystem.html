License Server Implementation Tutorial

This tutorial will guide you through implementing a complete license server for the GProd anti-piracy system.

Prerequisites

  • .NET 8 SDK
  • SQL Server or PostgreSQL
  • Azure account (optional, for Key Vault)
  • Visual Studio 2022 or VS Code

Part 3: Cryptography Utilities

Create GProdLicenseServer.Core/Security/CryptoHelper.cs:

using System.Security.Cryptography;
using System.Text;

namespace GProdLicenseServer.Core.Security;

public static class CryptoHelper
{
    public static string ComputeHMAC(string data, string secret)
    {
        using var hmac = new HMACSHA256(Encoding.UTF8.GetBytes(secret));
        var hash = hmac.ComputeHash(Encoding.UTF8.GetBytes(data));
        return Convert.ToBase64String(hash);
    }
    
    public static bool VerifyHMAC(string data, string signature, string secret)
    {
        var computed = ComputeHMAC(data, secret);
        return CryptographicOperations.FixedTimeEquals(
            Convert.FromBase64String(computed),
            Convert.FromBase64String(signature)
        );
    }
    
    public static string ComputeSHA256(this string input)
    {
        using var sha = SHA256.Create();
        var bytes = Encoding.UTF8.GetBytes(input);
        var hash = sha.ComputeHash(bytes);
        return Convert.ToBase64String(hash);
    }
    
    public static string GenerateLicenseKey()
    {
        var bytes = new byte[16];
        using var rng = RandomNumberGenerator.Create();
        rng.GetBytes(bytes);
        
        // Format as XXXX-XXXX-XXXX-XXXX
        var hex = Convert.ToHexString(bytes);
        return $"{hex[..4]}-{hex[4..8]}-{hex[8..12]}-{hex[12..16]}";
    }
}

using Microsoft.EntityFrameworkCore;
using GProdLicenseServer.Core.Models;

namespace GProdLicenseServer.Api.Data;

public class LicenseDbContext : DbContext
{
    public LicenseDbContext(DbContextOptions<LicenseDbContext> options)
        : base(options)
    {
    }
    
    public DbSet<License> Licenses => Set<License>();
    public DbSet<RegisteredDevice> RegisteredDevices => Set<RegisteredDevice>();
    
    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);
        
        // License configuration
        modelBuilder.Entity<License>(entity =>
        {
            entity.HasKey(e => e.Id);
            entity.HasIndex(e => e.LicenseKey).IsUnique();
            entity.Property(e => e.LicenseKey).HasMaxLength(100).IsRequired();
            entity.Property(e => e.CustomerEmail).HasMaxLength(255).IsRequired();
            entity.Property(e => e.CustomerName).HasMaxLength(255);
            entity.Property(e => e.CompanyName).HasMaxLength(255);
            entity.Property(e => e.RevocationReason).HasMaxLength(500);
            entity.Property(e => e.CreatedAt).HasDefaultValueSql("GETUTCDATE()");
        });
        
        // RegisteredDevice configuration
        modelBuilder.Entity<RegisteredDevice>(entity =>
        {
            entity.HasKey(e => e.Id);
            entity.HasIndex(e => e.LicenseId);
            entity.HasIndex(e => e.FingerprintHash);
            entity.Property(e => e.FingerprintHash).HasMaxLength(64).IsRequired();
            entity.Property(e => e.DeviceName).HasMaxLength(255);
            entity.Property(e => e.IpAddress).HasMaxLength(45);
            
            // Relationship
            entity.HasOne(e => e.License)
                .WithMany(l => l.RegisteredDevices)
                .HasForeignKey(e => e.LicenseId)
                .OnDelete(DeleteBehavior.Cascade);
        });
    }
}

Create GProdLicenseServer.Api/Services/LicenseService.cs:

using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Options;
using System.Text.Json;
using GProdLicenseServer.Core.DTOs;
using GProdLicenseServer.Core.Models;
using GProdLicenseServer.Core.Security;
using GProdLicenseServer.Api.Data;
using GProdLicenseServer.Api.Configuration;

namespace GProdLicenseServer.Api.Services;

public class LicenseService : ILicenseService
{
    private readonly LicenseDbContext _db;
    private readonly ILogger<LicenseService> _logger;
    private readonly SecuritySettings _settings;
    
    public LicenseService(
        LicenseDbContext db,
        ILogger<LicenseService> logger,
        IOptions<SecuritySettings> settings)
    {
        _db = db;
        _logger = logger;
        _settings = settings.Value;
    }
    
    public async Task<HeartbeatResponse> ValidateHeartbeatAsync(
        HeartbeatRequest request, 
        string? ipAddress)
    {
        try
        {
            // 1. Validate request signature
            if (!ValidateRequestSignature(request))
            {
                _logger.LogWarning(
                    "Invalid signature for license {LicenseKey} from IP {IP}",
                    request.LicenseKey[..8] + "...",
                    ipAddress);
                    
                return CreateErrorResponse("Invalid request signature");
            }
            
            // 2. Validate timestamp (prevent replay attacks)
            var timeDiff = Math.Abs((DateTime.UtcNow - request.ClientTimestamp).TotalMinutes);
            if (timeDiff > 5)
            {
                _logger.LogWarning(
                    "Timestamp drift too large ({Minutes} min) for license {LicenseKey}",
                    timeDiff,
                    request.LicenseKey[..8] + "...");
                    
                return CreateErrorResponse("Invalid timestamp");
            }
            
            // 3. Lookup license
            var license = await _db.Licenses
                .Include(l => l.RegisteredDevices)
                .FirstOrDefaultAsync(l => l.LicenseKey == request.LicenseKey);
            
            if (license == null)
            {
                _logger.LogWarning("License not found: {LicenseKey}", 
                    request.LicenseKey[..8] + "...");
                return CreateErrorResponse("Invalid license");
            }
            
            // 4. Check revocation
            if (license.IsRevoked)
            {
                _logger.LogInformation("Revoked license attempted: {LicenseKey}", 
                    request.LicenseKey[..8] + "...");
                return CreateErrorResponse($"License revoked: {license.RevocationReason}");
            }
            
            // 5. Check expiry
            if (license.ExpiryDate.HasValue && license.ExpiryDate.Value < DateTime.UtcNow)
            {
                _logger.LogInformation("Expired license attempted: {LicenseKey}", 
                    request.LicenseKey[..8] + "...");
                return CreateErrorResponse("License expired");
            }
            
            // 6. Verify hardware fingerprint
            var deviceHash = HashHardwareFingerprint(request.HardwareInfo);
            var device = license.RegisteredDevices
                .FirstOrDefault(d => d.FingerprintHash == deviceHash);
            
            if (device == null)
            {
                // New device - check limit
                if (license.RegisteredDevices.Count >= license.MaxDevices)
                {
                    _logger.LogWarning(
                        "Device limit exceeded for license {LicenseKey}. Current: {Current}, Max: {Max}",
                        request.LicenseKey[..8] + "...",
                        license.RegisteredDevices.Count,
                        license.MaxDevices);
                        
                    return CreateErrorResponse(
                        $"Device limit exceeded ({license.MaxDevices} devices maximum)");
                }
                
                // Register new device
                var hardwareInfo = JsonSerializer.Deserialize<HardwareFingerprint>(
                    request.HardwareInfo);
                
                device = new RegisteredDevice
                {
                    LicenseId = license.Id,
                    FingerprintHash = deviceHash,
                    FirstSeen = DateTime.UtcNow,
                    LastSeen = DateTime.UtcNow,
                    DeviceName = hardwareInfo?.MacAddress,
                    IpAddress = ipAddress
                };
                
                _db.RegisteredDevices.Add(device);
                
                _logger.LogInformation(
                    "New device registered for license {LicenseKey}. Total devices: {Count}",
                    request.LicenseKey[..8] + "...",
                    license.RegisteredDevices.Count + 1);
            }
            else
            {
                // Update existing device
                device.LastSeen = DateTime.UtcNow;
                device.IpAddress = ipAddress;
            }
            
            await _db.SaveChangesAsync();
            
            // 7. Generate successful response
            var response = new HeartbeatResponse
            {
                Valid = true,
                NextCheckIn = DateTime.UtcNow.AddHours(
                    24 + Random.Shared.Next(-2, 2)), // 22-26 hours
                ServerTimestamp = DateTime.UtcNow
            };
            
            response.Signature = SignResponse(response, request.LicenseKey);
            
            _logger.LogInformation(
                "Successful heartbeat for license {LicenseKey}",
                request.LicenseKey[..8] + "...");
            
            return response;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, 
                "Error processing heartbeat for license {LicenseKey}",
                request.LicenseKey);
            return CreateErrorResponse("Internal server error");
        }
    }
    
    public async Task<string> CreateLicenseAsync(
        string customerEmail,
        string? customerName,
        string? companyName,
        int maxDevices,
        DateTime? expiryDate)
    {
        var license = new License
        {
            LicenseKey = CryptoHelper.GenerateLicenseKey(),
            CustomerEmail = customerEmail,
            CustomerName = customerName,
            CompanyName = companyName,
            MaxDevices = maxDevices,
            ExpiryDate = expiryDate,
            CreatedAt = DateTime.UtcNow,
            IsRevoked = false
        };
        
        _db.Licenses.Add(license);
        await _db.SaveChangesAsync();
        
        _logger.LogInformation(
            "Created license {LicenseKey} for customer {Email}",
            license.LicenseKey,
            customerEmail);
        
        return license.LicenseKey;
    }
    
    public async Task<bool> RevokeLicenseAsync(string licenseKey, string reason)
    {
        var license = await _db.Licenses
            .FirstOrDefaultAsync(l => l.LicenseKey == licenseKey);
        
        if (license == null)
            return false;
        
        license.IsRevoked = true;
        license.RevocationReason = reason;
        
        await _db.SaveChangesAsync();
        
        _logger.LogWarning(
            "License {LicenseKey} revoked. Reason: {Reason}",
            licenseKey,
            reason);
        
        return true;
    }
    
    public async Task<License?> GetLicenseInfoAsync(string licenseKey)
    {
        return await _db.Licenses
            .Include(l => l.RegisteredDevices)
            .FirstOrDefaultAsync(l => l.LicenseKey == licenseKey);
    }
    
    // Private helper methods
    
    private bool ValidateRequestSignature(HeartbeatRequest request)
    {
        var dataToSign = $"{request.LicenseKey}|{request.HardwareInfo}|{request.ClientTimestamp:O}";
        return CryptoHelper.VerifyHMAC(dataToSign, request.Signature, _settings.SharedSecret);
    }
    
    private string SignResponse(HeartbeatResponse response, string licenseKey)
    {
        var dataToSign = $"{response.Valid}|{response.NextCheckIn:O}|{response.ServerTimestamp:O}|{licenseKey}";
        return CryptoHelper.ComputeHMAC(dataToSign, _settings.SharedSecret);
    }
    
    private string HashHardwareFingerprint(string hardwareInfoJson)
    {
        var hw = JsonSerializer.Deserialize<HardwareFingerprint>(hardwareInfoJson);
        if (hw == null)
            throw new ArgumentException("Invalid hardware fingerprint");
        
        return hw.ToNormalizedString().ComputeSHA256();
    }
    
    private HeartbeatResponse CreateErrorResponse(string reason)
    {
        return new HeartbeatResponse
        {
            Valid = false,
            Reason = reason,
            NextCheckIn = DateTime.UtcNow.AddHours(1), // Retry in 1 hour
            ServerTimestamp = DateTime.UtcNow,
            Signature = string.Empty
        };
    }
}

Part 7: API Controllers

Create GProdLicenseServer.Api/Controllers/LicenseController.cs:

using Microsoft.AspNetCore.Mvc;
using GProdLicenseServer.Core.DTOs;
using GProdLicenseServer.Api.Services;

namespace GProdLicenseServer.Api.Controllers;

[ApiController]
[Route("api/[controller]")]
public class LicenseController : ControllerBase
{
    private readonly ILicenseService _licenseService;
    private readonly ILogger<LicenseController> _logger;
    
    public LicenseController(
        ILicenseService licenseService,
        ILogger<LicenseController> logger)
    {
        _licenseService = licenseService;
        _logger = logger;
    }
    
    [HttpPost("heartbeat")]
    public async Task<ActionResult<HeartbeatResponse>> Heartbeat(
        [FromBody] HeartbeatRequest request)
    {
        var ipAddress = HttpContext.Connection.RemoteIpAddress?.ToString();
        var response = await _licenseService.ValidateHeartbeatAsync(request, ipAddress);
        
        if (!response.Valid)
        {
            return Ok(response); // Return 200 with Valid=false
        }
        
        return Ok(response);
    }
    
    [HttpPost("create")]
    public async Task<ActionResult<CreateLicenseResponse>> CreateLicense(
        [FromBody] CreateLicenseRequest request)
    {
        // In production, add authentication/authorization here
        // For now, this is an admin endpoint
        
        if (string.IsNullOrWhiteSpace(request.CustomerEmail))
            return BadRequest("Customer email is required");
        
        if (request.MaxDevices < 1)
            return BadRequest("MaxDevices must be at least 1");
        
        var licenseKey = await _licenseService.CreateLicenseAsync(
            request.CustomerEmail,
            request.CustomerName,
            request.CompanyName,
            request.MaxDevices,
            request.ExpiryDate
        );
        
        return Ok(new CreateLicenseResponse
        {
            LicenseKey = licenseKey,
            CreatedAt = DateTime.UtcNow
        });
    }
    
    [HttpPost("revoke")]
    public async Task<ActionResult> RevokeLicense(
        [FromBody] RevokeLicenseRequest request)
    {
        // In production, add authentication/authorization here
        
        if (string.IsNullOrWhiteSpace(request.LicenseKey))
            return BadRequest("License key is required");
        
        var success = await _licenseService.RevokeLicenseAsync(
            request.LicenseKey,
            request.Reason ?? "No reason provided"
        );
        
        if (!success)
            return NotFound("License not found");
        
        return Ok(new { Message = "License revoked successfully" });
    }
    
    [HttpGet("info/{licenseKey}")]
    public async Task<ActionResult> GetLicenseInfo(string licenseKey)
    {
        // In production, add authentication/authorization here
        
        var license = await _licenseService.GetLicenseInfoAsync(licenseKey);
        
        if (license == null)
            return NotFound();
        
        return Ok(new
        {
            license.LicenseKey,
            license.CustomerEmail,
            license.CustomerName,
            license.CompanyName,
            license.CreatedAt,
            license.ExpiryDate,
            license.IsRevoked,
            license.RevocationReason,
            license.MaxDevices,
            DeviceCount = license.RegisteredDevices.Count,
            Devices = license.RegisteredDevices.Select(d => new
            {
                d.DeviceName,
                d.FirstSeen,
                d.LastSeen,
                d.IpAddress
            })
        });
    }
}

#!/bin/bash

az appservice plan create \
  --name "${APP_NAME}-plan" \

  --sku B1

# Create Web App
az webapp create \
  --name $APP_NAME \
  --resource-group $RESOURCE_GROUP \

Part 11: Security Hardening

Update Program.cs:

// Add before builder.Build()
if (builder.Environment.IsProduction())
{
    var keyVaultUrl = builder.Configuration["KeyVaultUrl"];
