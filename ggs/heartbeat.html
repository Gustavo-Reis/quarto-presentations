Overview

Software piracy protection system combining:

  • Hardware fingerprinting
  • License server validation
  • Periodic heartbeat mechanism
  • Cryptographic security

This presentation covers the complete implementation of an anti-piracy system for GProd, our Blazor-based product.

Device Registration

// Verify hardware fingerprint
var deviceHash = HashHardwareFingerprint(request.HardwareInfo);

var device = license.RegisteredDevices
    .FirstOrDefault(d => d.FingerprintHash == deviceHash);

if (device == null)
{
    // New device - check limit
    if (license.RegisteredDevices.Count >= license.MaxDevices)
        return Ok(new { 
            Valid = false, 
            Reason = "Device limit exceeded" 
        });
    
    // Register new device
    license.RegisteredDevices.Add(new RegisteredDevice {
        FingerprintHash = deviceHash,
        FirstSeen = DateTime.UtcNow
    });
}

Client Implementation

Background Service

public class LicenseHeartbeatService : BackgroundService
{
    private const int GracePeriodDays = 7;
    
    protected override async Task ExecuteAsync(
        CancellationToken stoppingToken)
    {
        while (!stoppingToken.IsCancellationRequested)
        {
            try
            {
                var result = await PerformHeartbeat();
                
                if (!result.Valid)
                    await HandleInvalidLicense(result.Reason);
                else
                    SaveLastSuccessfulHeartbeat(result.ServerTimestamp);
                
                await Task.Delay(CalculateDelay(result), stoppingToken);
            }
            catch
            {
                await HandleNetworkFailure();
            }
        }
    }
}

Heartbeat Execution

private async Task<HeartbeatResponse> PerformHeartbeat()
{
    var request = new HeartbeatRequest
    {
        LicenseKey = _validator.GetStoredLicense(),
        HardwareInfo = JsonSerializer.Serialize(
            _validator.GetHardwareFingerprint()),
        ClientTimestamp = DateTime.UtcNow
    };
    
    // Sign request
    request.Signature = ComputeHMAC(request);
    
    var response = await _client.PostAsJsonAsync(
        "https://server.com/api/license/heartbeat", 
        request);
    
    var result = await response.Content
        .ReadFromJsonAsync<HeartbeatResponse>();
    
    // Verify response signature
    if (!VerifyResponseSignature(result))
        throw new SecurityException("Invalid response");
    
    return result;
}

Local Storage

private void SaveLastSuccessfulHeartbeat(DateTime serverTime)
{
    // Use hardware fingerprint as encryption key
    var hwKey = _validator.GetHardwareFingerprint()
        .ComputeSHA256Bytes();
    
    // Encrypt timestamp
    var encrypted = CryptoHelper.AesEncrypt(
        serverTime.ToString("O"), 
        hwKey);
    
    // Store in registry
    RegistryHelper.SetEncryptedValue("LastHeartbeat", encrypted);
}

Encryption with hardware key means timestamp becomes invalid if hardware changes

Grace Period Validation

private bool IsWithinGracePeriod()
{
    try
    {
        // Decrypt with hardware key
        var hwKey = _validator.GetHardwareFingerprint()
            .ComputeSHA256Bytes();
        
        var encrypted = RegistryHelper
            .GetEncryptedValue("LastHeartbeat");
        
        var lastHeartbeat = DateTime.Parse(
            CryptoHelper.AesDecrypt(encrypted, hwKey));
        
        return (DateTime.UtcNow - lastHeartbeat).TotalDays 
            <= GracePeriodDays;
    }
    catch
    {
        return false; // No valid heartbeat stored
    }
}

Network Failure Handling

private async Task HandleNetworkFailure()
{
    // Check if still within grace period
    if (!IsWithinGracePeriod())
    {
        await HandleInvalidLicense(
            "Unable to verify license - grace period expired");
    }
    
    // Retry in 1 hour
    await Task.Delay(TimeSpan.FromHours(1));
}

Cryptography

HMAC Signatures

public static string ComputeHMAC(string data, string secret)
{
    using var hmac = new HMACSHA256(
        Encoding.UTF8.GetBytes(secret));
    
    var hash = hmac.ComputeHash(
        Encoding.UTF8.GetBytes(data));
    
    return Convert.ToBase64String(hash);
}

public static bool VerifyHMAC(string data, string signature)
{
    var computed = ComputeHMAC(data, secret);
    
    return CryptographicOperations.FixedTimeEquals(
        Convert.FromBase64String(computed),
        Convert.FromBase64String(signature)
    );
}

Fixed-time comparison prevents timing attacks

AES Encryption

public static string AesEncrypt(string plainText, byte[] key)
{
    using var aes = Aes.Create();
    aes.Key = DeriveKey(key, 32); // 256-bit key
    aes.GenerateIV();
    
    using var encryptor = aes.CreateEncryptor();
    using var ms = new MemoryStream();
    
    // Prepend IV to ciphertext
    ms.Write(aes.IV, 0, aes.IV.Length);
    
    using (var cs = new CryptoStream(ms, encryptor, CryptoStreamMode.Write))
    using (var sw = new StreamWriter(cs))
    {
        sw.Write(plainText);
    }
    
    return Convert.ToBase64String(ms.ToArray());
}

AES Decryption

public static string AesDecrypt(string cipherText, byte[] key)
{
    var fullCipher = Convert.FromBase64String(cipherText);
    
    using var aes = Aes.Create();
    aes.Key = DeriveKey(key, 32);
    
    // Extract IV from beginning
    var iv = new byte[aes.BlockSize / 8];
    var cipher = new byte[fullCipher.Length - iv.Length];
    
    Array.Copy(fullCipher, iv, iv.Length);
    Array.Copy(fullCipher, iv.Length, cipher, 0, cipher.Length);
    
    aes.IV = iv;
    
    using var decryptor = aes.CreateDecryptor();
    using var ms = new MemoryStream(cipher);
    using var cs = new CryptoStream(ms, decryptor, CryptoStreamMode.Read);
    using var sr = new StreamReader(cs);
    
    return sr.ReadToEnd();
}

Key Derivation

private static byte[] DeriveKey(byte[] input, int keyLength)
{
    using var sha = SHA256.Create();
    var hash = sha.ComputeHash(input);
    
    if (keyLength <= hash.Length)
        return hash.Take(keyLength).ToArray();
    
    // For longer keys, use PBKDF2
    using var pbkdf2 = new Rfc2898DeriveBytes(
        input, 
        saltBytes,
        iterations: 10000, 
        HashAlgorithmName.SHA256
    );
    
    return pbkdf2.GetBytes(keyLength);
}

  • License key sharing
  • Offline cracking attempts
  • Network interception (MITM)
  • Local timestamp tampering

  • Automated piracy tools

Key sharing Hardware fingerprinting + device limits

Best Practices

Security Checklist

Operational Guidelines

Grace period:

  • 7 days for typical usage
  • Longer for industries with poor connectivity
  • Shorter for high-security requirements

Heartbeat interval:

  • 24 hours ± 2 hours (jitter)
  • Too frequent = server load
  • Too infrequent = delayed revocation

Device limits:

  • 1-3 for individual licenses
  • 5-10 for team licenses
  • Unlimited for enterprise

Appendix: Hardware Fingerprint

public class HardwareFingerprint
{
    public string CpuId { get; set; }
    public string MacAddress { get; set; }
    public string DiskSerial { get; set; }
    public string MotherboardSerial { get; set; }
    
    public static HardwareFingerprint Generate()
    {
        return new HardwareFingerprint
        {
            CpuId = GetCpuId(),
            MacAddress = GetMacAddress(),
            DiskSerial = GetDiskSerial(),
            MotherboardSerial = GetMotherboardSerial()
        };
    }
    
    private static string GetCpuId()
    {
        using var searcher = new ManagementObjectSearcher(
            "SELECT ProcessorId FROM Win32_Processor");
        return searcher.Get().Cast<ManagementObject>()
            .First()["ProcessorId"].ToString();
    }
    
    // Similar methods for MAC, disk, motherboard...
}

Appendix: Rate Limiting

public class RateLimitMiddleware
{
    private static readonly ConcurrentDictionary<string, RateLimitInfo> 
        _requests = new();
    
    public async Task InvokeAsync(HttpContext context, RequestDelegate next)
    {
        var ip = context.Connection.RemoteIpAddress?.ToString();
        var info = _requests.GetOrAdd(ip, _ => new RateLimitInfo());
        
        if (!info.AllowRequest())
        {
            context.Response.StatusCode = 429; // Too Many Requests
            await context.Response.WriteAsync(
                "Rate limit exceeded. Try again later.");
            return;
        }
        
        await next(context);
    }
}

public class RateLimitInfo
{
    private Queue<DateTime> _requests = new();
    private const int MaxRequests = 100;
    private static readonly TimeSpan Window = TimeSpan.FromHours(1);
    
    public bool AllowRequest()
    {
        var now = DateTime.UtcNow;
        _requests = new Queue<DateTime>(
            _requests.Where(t => now - t < Window));
        
        if (_requests.Count >= MaxRequests)
            return false;
        
        _requests.Enqueue(now);
        return true;
    }
}
